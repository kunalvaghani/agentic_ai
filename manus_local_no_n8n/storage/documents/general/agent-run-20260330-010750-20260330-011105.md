```python
import tkinter as tk
from tkinter import messagebox

class CalculatorApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Python Calculator")
        self.root.geometry("350x450")
        self.root.resizable(False, False)
        
        # Variable to store the expression
        self.expression = ""
        
        # Create the display screen
        self.display = tk.Entry(root, font=('Arial', 24), bd=10, bg='#f0f0f0', justify='right')
        self.display.grid(row=0, column=0, columnspan=4, padx=10, pady=10, sticky='nsew')
        
        # Configure grid weights for resizing
        for i in range(5):
            self.root.grid_rowconfigure(i, weight=1)
            self.root.grid_columnconfigure(i, weight=1)
            
        # Button configuration
        buttons = [
            ('7', 1, 0), ('8', 1, 1), ('9', 1, 2), ('/', 1, 3),
            ('4', 2, 0), ('5', 2, 1), ('6', 2, 2), ('*', 2, 3),
            ('1', 3, 0), ('2', 3, 1), ('3', 3, 2), ('-', 3, 3),
            ('0', 4, 0), ('.', 4, 1), ('=', 4, 2), ('+', 4, 3),
            ('C', 5, 0, 4), # Clear button spanning 2 columns
        ]
        
        # Create buttons
        for text, row, col, *span in buttons:
            if span:
                width = span[0]
                btn = tk.Button(root, text=text, font=('Arial', 18), bd=4, width=width, command=lambda t=text: self.on_button_click(t))
            else:
                btn = tk.Button(root, text=text, font=('Arial', 18), bd=4, command=lambda t=text: self.on_button_click(t))
            
            btn.grid(row=row, column=col, sticky='nsew')
            
        # Style buttons
        for widget in self.root.winfo_children():
            if isinstance(widget, tk.Button):
                widget.config(bg='#e0e0e0')
                if widget.cget('text') in ['/', '*', '-', '+', '=']:
                    widget.config(bg='#ffcc00') # Operator color
                elif widget.cget('text') == 'C':
                    widget.config(bg='#ff5555') # Clear color

    def on_button_click(self, char):
        if char == 'C':
            self.expression = ""
            self.display.delete(0, tk.END)
        elif char == '=':
            try:
                # Evaluate the expression safely
                # We replace visual symbols with python operators if needed, 
                # but standard input usually works. 
                # Note: eval is used here for simplicity in a local tool.
                # In production, a custom parser is preferred.
                result = str(eval(self.expression))
                self.display.delete(0, tk.END)
                self.display.insert(0, result)
                self.expression = result
            except ZeroDivisionError:
                self.display.insert(0, "Error")
                self.expression = ""
            except SyntaxError:
                self.display.insert(0, "Syntax Error")
                self.expression = ""
            except Exception:
                self.display.insert(0, "Error")
                self.expression = ""
        else:
            self.expression += str(char)
            self.display.delete(0, tk.END)
            self.display.insert(0, self.expression)

if __name__ == "__main__":
    root = tk.Tk()
    app = CalculatorApp(root)
    root.mainloop()
```
