```python
import tkinter as tk
from tkinter import messagebox

class CalculatorApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Python Calculator")
        self.root.geometry("350x450")
        self.root.resizable(False, False)
        
        # Variable to store the current expression
        self.expression = ""
        
        # Create the display screen
        self.display = tk.Entry(root, font=('Arial', 24), bd=10, bg="#2c3e50", fg="#fff", justify='right')
        self.display.grid(row=0, column=0, columnspan=4, padx=10, pady=10, sticky="nsew")
        
        # Configure grid weights for resizing
        for i in range(5):
            self.root.grid_rowconfigure(i, weight=1)
            self.root.grid_columnconfigure(i, weight=1)
            
        # Button configuration
        buttons = [
            ('7', 1, 0), ('8', 1, 1), ('9', 1, 2), ('/', 1, 3),
            ('4', 2, 0), ('5', 2, 1), ('6', 2, 2), ('*', 2, 3),
            ('1', 3, 0), ('2', 3, 1), ('3', 3, 2), ('-', 3, 3),
            ('0', 4, 0), ('.', 4, 1), ('C', 4, 2), ('=', 4, 3),
        ]
        
        # Create buttons
        for text, row, col in buttons:
            self.create_button(text, row, col)
            
        # Add Enter key support
        self.root.bind('<Return>', lambda event: self.calculate())

    def create_button(self, text, row, col):
        button = tk.Button(self.root, text=text, font=('Arial', 18), bd=4, fg="white", bg="#34495e", activebackground="#2c3e50", cursor="hand2")
        button.grid(row=row, column=col, padx=5, pady=5, sticky="nsew")
        button.bind("<ButtonRelease-1>", lambda e, t=text: self.on_button_click(t))
        return button

    def on_button_click(self, char):
        if char == 'C':
            self.expression = ""
            self.display.delete(0, tk.END)
        elif char == '=':
            self.calculate()
        else:
            self.expression += char
            self.display.delete(0, tk.END)
            self.display.insert(0, self.expression)

    def calculate(self):
        try:
            # Replace visual symbols with Python operators if needed (already handled in input)
            # Evaluate the expression safely
            # We use eval here but restrict input to numbers and operators via UI
            result = eval(self.expression)
            
            # Handle division by zero or other float issues
            if isinstance(result, float):
                if result.is_integer():
                    self.display.delete(0, tk.END)
                    self.display.insert(0, str(int(result)))
                else:
                    self.display.delete(0, tk.END)
                    self.display.insert(0, str(result))
            else:
                self.display.delete(0, tk.END)
                self.display.insert(0, str(result))
        except ZeroDivisionError:
            messagebox.showerror("Error", "Cannot divide by zero!")
            self.expression = ""
            self.display.delete(0, tk.END)
        except SyntaxError:
            messagebox.showerror("Error", "Invalid Expression!")
            self.expression = ""
            self.display.delete(0, tk.END)
        except Exception as e:
            messagebox.showerror("Error", f"Error: {e}")
            self.expression = ""
            self.display.delete(0, tk.END)

if __name__ == "__main__":
    root = tk.Tk()
    app = CalculatorApp(root)
    root.mainloop()
```
