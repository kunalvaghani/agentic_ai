```python
import tkinter as tk
from tkinter import messagebox, ttk
import re

class ElegantCalculator:
    def __init__(self, root):
        self.root = root
        self.root.title("Elegant Calculator")
        self.root.geometry("320x480")
        self.root.resizable(False, False)
        
        # Configure style
        self.root.configure(bg="#2b2b2b")
        
        # Variables
        self.expression = ""
        self.history = []
        
        # Create UI components
        self.create_display()
        self.create_buttons()
        
        # Bind keyboard events
        self.root.bind('<Key>', self.on_key_press)

    def create_display(self):
        # Main display frame
        display_frame = tk.Frame(self.root, bg="#2b2b2b", pady=10)
        display_frame.pack(fill=tk.X, padx=10)
        
        # Entry widget for expression
        self.display = tk.Entry(
            display_frame, 
            font=('Segoe UI', 24, 'bold'), 
            bg="#3c3f41", 
            fg="white",
            insertbackground="white",
            justify='right',
            bd=0,
            relief='flat',
            width=20
        )
        self.display.pack(fill=tk.X, padx=5, pady=5)
        self.display.insert(0, "0")
        
        # History label
        self.history_label = tk.Label(
            display_frame, 
            text="", 
            font=('Segoe UI', 10), 
            fg="#888888", 
            bg="#2b2b2b"
        )
        self.history_label.pack(side=tk.BOTTOM, fill=tk.X, padx=5, pady=2)

    def create_buttons(self):
        # Button configuration
        btn_config = {
            'font': ('Segoe UI', 14),
            'bg': '#3c3f41',
            'fg': 'white',
            'bd': 0,
            'relief': 'flat',
            'cursor': 'hand2',
            'activebackground': '#4a4d50',
            'activeforeground': 'white'
        }
        
        # Grid layout
        # Row 1: Clear, Backspace, %, /
        # Row 2: 7, 8, 9, *
        # Row 3: 4, 5, 6, -
        # Row 4: 1, 2, 3, +
        # Row 5: 0, ., =, ( )
        
        buttons = [
            ('C', btn_config, self.clear_all),
            ('⌫', btn_config, self.backspace),
            ('%', btn_config, self.percentage),
            ('/', btn_config, self.append_operator, '/'),
            
            ('7', btn_config, self.append_number, '7'),
            ('8', btn_config, self.append_number, '8'),
            ('9', btn_config, self.append_number, '9'),
            ('*', btn_config, self.append_operator, '*'),
            
            ('4', btn_config, self.append_number, '4'),
            ('5', btn_config, self.append_number, '5'),
            ('6', btn_config, self.append_number, '6'),
            ('-', btn_config, self.append_operator, '-'),
            
            ('1', btn_config, self.append_number, '1'),
            ('2', btn_config, self.append_number, '2'),
            ('3', btn_config, self.append_number, '3'),
            ('+', btn_config, self.append_operator, '+'),
            
            ('0', btn_config, self.append_number, '0'),
            ('.', btn_config, self.append_number, '.'),
            ('=', {'bg': '#4caf50', 'fg': 'white'}, self.calculate),
            ('(', btn_config, self.append_operator, '('),
            (')', btn_config, self.append_operator, ')'),
        ]
        
        for i, (text, config, command, *args) in enumerate(buttons):
            row = i // 4
            col = i % 4
            
            # Special handling for the equals button (spanning 2 columns)
            if text == '=':
                btn = tk.Button(
                    self.root, 
                    text=text, 
                    command=command,
                    **config
                )
                btn.grid(row=row, column=col, columnspan=2, sticky='nsew', padx=5, pady=5)
            else:
                btn = tk.Button(
                    self.root, 
                    text=text, 
                    command=lambda t=text, c=config, cmd=command, arg=args[0] if args else None: self._make_click(t, c, cmd, arg),
                    **config
                )
                btn.grid(row=row, column=col, sticky='nsew', padx=5, pady=5)
            
            self.root.grid_rowconfigure(row, weight=1)
            self.root.grid_columnconfigure(col, weight=1)

    def _make_click(self, text, config, command, arg=None):
        """Helper to bind button click with dynamic arguments"""
        def on_click(event=None):
            if arg:
                command(self, arg)
            else:
                command(self)
        return on_click

    def clear_all(self):
        self.expression = ""
        self.display.delete(0, tk.END)
        self.display.insert(0, "0")
        self.history_label.config(text="")

    def backspace(self):
        current = self.display.get()
        if current == "0":
            return
        new_val = current[:-1]
        if new_val == "":
            new_val = "0"
        self.display.delete(0, tk.END)
        self.display.insert(0, new_val)

    def append_number(self, num):
        current = self.display.get()
        if current == "0" or current == "Error":
            self.display.delete(0, tk.END)
        self.display.insert(tk.END, num)

    def append_operator(self, op):
        current = self.display.get()
        if current == "Error":
            return
        # Prevent multiple operators in a row
        if re.search(r'[+\-*/%]', current):
            # Replace last operator if exists
            self.display.delete(0, tk.END)
            self.display.insert(0, current[:-1] + op)
        else:
            self.display.insert(tk.END, op)

    def percentage(self):
        current = self.display.get()
        if current == "Error":
            return
        try:
            val = float(current)
            new_val = val / 100
            self.display.delete(0, tk.END)
            self.display.insert(0, str(new_val))
        except ValueError:
            self.show_error()

    def calculate(self):
        current = self.display.get()
        if current == "Error":
            return
        
        try:
            # Replace visual operators with python operators
            # Note: Python handles precedence automatically
            safe_expr = current.replace('×', '*').replace('÷', '/')
            
            # Evaluate safely
            # We use eval but restrict characters to be safe
            if not re.match(r'^[\d+\-*/%.() ]+$', safe_expr):
                raise ValueError("Invalid characters")
                
            result = eval(safe_expr)
            
            # Format result
            if isinstance(result, float):
                if result.is_integer():
                    result = int(result)
                else:
                    result = round(result, 10) # Limit decimal places
            
            self.display.delete(0, tk.END)
            self.display.insert(0, str(result))
            
            # Update history
            self.history.append(f"{current} = {result}")
            self.history_label.config(text=self.history[-1])
            
        except ZeroDivisionError:
            self.display.insert(0, "Error")
        except Exception:
            self.display.insert(0, "Error")

    def show_error(self):
        self.display.insert(0, "Error")

    def on_key_press(self, event):
        key = event.char
        if key.isdigit():
            self.append_number(key)
        elif key in ['+', '-', '*', '/', '%', '(', ')']:
            self.append_operator(key)
        elif key == '.':
            self.append_number('.')
        elif key == 'BackSpace':
            self.backspace()
        elif key == 'c' or key == 'C':
            self.clear_all()
        elif key == 'Escape':
            self.clear_all()
        elif key == '=':
            self.calculate()

if __name__ == "__main__":
    root = tk.Tk()
    app = ElegantCalculator(root)
    root.mainloop()
```

### How to Run
1.  Save the code above as `calculator.py`.
2.  Run it in your terminal:
    ```bash

    ```

### Features Included
*   **Elegant UI**: Dark theme with high contrast, rounded corners (via flat styling), and a green "Equals" button for visual hierarchy.
*   **Keyboard Support**: You can type numbers, operators, and use `Backspace` or `Escape` to clear.
*   **History**: Shows the last calculation result at the bottom of the display.
*   **Error Handling**: Displays "Error" for division by zero or invalid syntax.
*   **Responsive Layout**: Uses a grid system that adapts to the fixed window size.

No external libraries are required; this runs on any standard Python installation.
